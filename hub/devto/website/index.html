Python Example

from datasets import load_dataset

# Load the dataset
dataset = load_dataset("Alaamer/devto-articles-dataset")

# Access the data
for article in dataset['train']:
    print(f"Title: {article['title']}")
    print(f"Author: {article['author']['username']}")
    print(f"Reactions: {article['reactions']['total']}")
    print("---")
